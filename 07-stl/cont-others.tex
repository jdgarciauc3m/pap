\section{Otras operaciones}

\begin{frame}[t]{Otras operaciones}
\begin{itemize}
  \item Comparación:
    \begin{itemize}
      \item \cppid{c1==c2}: Iguales si hay igualdad elemento a elemento.
      \item \cppid{c1<c2}: Menor-que aplicando orden lexicográfico.
      \item Resto (\cppid{<=}, \cppid{>}, \cppid{>=}, \cppid{!=}) derivados de estos dos.
    \end{itemize}
  \vfill\pause
  \item Intercambio:
    \begin{itemize}
      \item \cppid{c1.swap(c2)}: Intercambia los valores de forma \cppkey{noexcept}.
      \item \cppid{swap(c1,c2)}: Equivalente al anterior.
    \end{itemize}
\end{itemize}
\end{frame}
